#include <stdio.h>

int main()
{
    int arrayNum[3][3] = {{9, 1, 1},
                          {0, 6, 9},
                          {6, 7, 0}};

    for (int i = 0; i < 3; i++)
    {
        for (int j = 0; j < 3; j++)
        {
            printf("%d ", arrayNum[i][j]);
        }
        printf("\n");
    }
}
