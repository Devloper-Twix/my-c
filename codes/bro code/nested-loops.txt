#include <stdio.h>

int main()
{
    for (int i = 0; i < 5; i++)
    {
        printf("%d ->\t", i);
        for (int j = 1; j < 6; j++)
        {
            printf("%d\t", j);
        }
        printf("\n");
    }

    printf("\nDone\n\n");

    for (int i = 0; i < 3; i++)
    {
        int j = 0;
        while (j < 10)
        {
            j++;
            printf("%d ", j);
        }
        printf("\n");
    }

    return 0;
}
