#include <stdio.h>
#include <stdlib.h>
#include <math.h>

#define PI 3.14159265358979323846

int main()
{
    double radius, area, circumference, surfaceArea, volume;

    printf("Enter radius:\t");
    scanf("%lf", &radius);

    area = PI * radius * radius;
    circumference = 2 * PI * radius;
    surfaceArea = 4 * PI * radius * radius;
    volume = (4.0 / 3.0) * PI * pow(radius, 3);

    printf("Area:\t\t%.2lf\nCircumference:\t%.2lf\nSurface Area:\t%.2lf\nVolume:\t\t%.2lf\n", area, circumference, surfaceArea, volume);
    printf("All correct up to 2dp");
    return 0;
}
