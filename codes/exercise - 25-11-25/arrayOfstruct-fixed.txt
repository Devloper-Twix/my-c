#include <stdio.h>

typedef struct
{
  char name[20];
  int health;
  int score;
} player;

int maxScore(int score[], int n);

int main()
{
  player players[5] = {
      {"asyed21", 50, 200},
      {"red", 20, 60},
      {"pussyDestroyer6969", 40, 140},
      {"sheAintREAL34", 70, 800},
      {"nahi", 100, 500}};

  int scores[5];
  for (int i = 0; i < 5; i++)
    scores[i] = players[i].score;

  printf("%d", maxScore(scores, 5));

  return 0;
}

int maxScore(int score[], int n)
{
  int max = score[0];
  for (int i = 1; i < n; i++)
    if (score[i] > max)
      max = score[i];

  return max;
}
