#include <stdio.h>

int *findMax(int arr[], int n);

int main()
{
  int arr[5] = {5, 6, 2, 1};
  int numOfElements = sizeof(arr) / sizeof(arr[0]);
  int *max = findMax(arr, numOfElements);

  printf("Pointer: %d\nValue: %d", max, *max);

  return 0;
}

int *findMax(int arr[], int n)
{
  int temp = 0;
  int *pTemp = &temp;
  for (int i = 0; i < n; i++)
  {
    if (arr[i] > temp)
    {
      temp = arr[i];
    }
  }
  return pTemp;
}
