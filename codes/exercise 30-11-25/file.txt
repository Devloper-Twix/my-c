#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main()
{
  FILE *pF1 = fopen("note.txt", "w");
  char line[1024];
  if (pF1 == NULL)
  {
    perror("error");
  }
  printf("Now thou mayst scribe thy poesy: ");
  fgets(line, 1024, stdin);
  size_t len = strlen(line);
  if (len > 0)
  {
    line[len - 1] = '\0';
  }
  fprintf(pF1, "%s", line);
  fclose(pF1);

  FILE *pF2 = fopen("note.txt", "r");
  char buff[1024];
  if (pF2 == NULL)
  {
    perror("error");
  }
  printf("\nWhat I read\n\n");
  while (fgets(buff, 1024, pF2) != NULL)
  {
    printf("%s", buff);
  }
  fclose(pF2);

  return 0;
}
